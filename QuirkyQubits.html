# Submit a circuit to QuirkyQubits Cloud
from quirkyqubits import Circuit, cloud

c = Circuit(qubits=4)
c.entangle(0,1)
c.entangle(2,3)

job = cloud.submit(
    circuit=c,
    backend="simulator_xlarge",
    shots=8192
)

print(job.result())
